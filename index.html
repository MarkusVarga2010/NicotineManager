create table eigene_schulden (
  id bigint generated always as identity primary key,
  user_id uuid not null,
  glaeubiger text not null,
  betrag numeric not null,
  notiz text,
  bezahlt boolean not null default false,
  created_at timestamptz not null default now(),
  bezahlt_am timestamptz
);
alter table eigene_schulden enable row level security;
create policy "eigene schulden eigener user"
  on eigene_schulden for all
  using (auth.uid() = user_id)
  with check (auth.uid() = user_id);

create table produkt_varianten (
  id bigint generated always as identity primary key,
  user_id uuid not null,
  produkt_id text not null,
  name text not null,
  preis numeric not null default 0,
  einkaufspreis numeric not null default 0,
  bild_url text,
  created_at timestamptz not null default now()
);
alter table produkt_varianten enable row level security;
create policy "varianten lesen"
  on produkt_varianten for select
  using (true);
create policy "varianten eigene anlegen"
  on produkt_varianten for insert
  with check (auth.uid() = user_id);
create policy "varianten eigene aktualisieren"
  on produkt_varianten for update
  using (auth.uid() = user_id);
create policy "varianten eigene loeschen"
  on produkt_varianten for delete
  using (auth.uid() = user_id);
